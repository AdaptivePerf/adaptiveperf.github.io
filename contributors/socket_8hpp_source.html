223 std::unique_ptr<Connection> accept(unsigned int buf_size) {
224 if (this->max_accepted != UNLIMITED_ACCEPTED &&
225 this->accepted >= this->max_accepted) {
226 throw std::runtime_error("Maximum accepted connections reached.");
227 }
228
229 std::unique_ptr<Connection> connection = this->accept_connection(buf_size);
230 this->accepted++;
231
232 return connection;
233 }
234
235 virtual ~Acceptor() { }
