211 std::unique_ptr<Connection> accept(unsigned int buf_size) {
212 if (this->max_accepted != UNLIMITED_ACCEPTED &&
213 this->accepted >= this->max_accepted) {
214 throw std::runtime_error("Maximum accepted connections reached.");
215 }
216
217 std::unique_ptr<Connection> connection = this->accept_connection(buf_size);
218 this->accepted++;
219
220 return connection;
221 }
